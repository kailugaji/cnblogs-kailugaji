Question：

Given nums = [2, 7, 11, 15], target = 9,

Because nums[0] + nums[1] = 2 + 7 = 9,

return [0, 1].

Answer：

int* twoSum(int* nums, int numsSize, int target) {
int j, i;
int *test = NULL;
test = (int )malloc(2*sizeof(int));
 for (i=0; i<numsSize ;i++){
  for (j=i+1; j<numsSize ;j++){
   if(nums[i]+nums[j] == target){
    *(test) = i;
    *(test+1) = j;
    }
  }
 }
 return test;
}
