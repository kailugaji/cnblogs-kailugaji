2.矩阵的乘法运算

 1 #include<stdio.h>
 2 #define M 10
 3 void main(){
 4     long int a[M][M], b[M][M], c[M][M]={0};
 5     int m, n, p, i, j, k;
 6 
 7     printf("Please input 3 numbers:\n");
 8     scanf("%d %d %d",&m, &n, &p);
 9 
10     printf("Please input array A (%d * %d)\n", m, n);
11     for(i=0;i<m;i++){
12         for(j=0;j<n;j++){
13             scanf("%ld", &a[i][j]);
14         }
15     }
16 
17     printf("Plaese input array B (%d * %d)\n", n, p);
18     for(i=0;i<n;i++){
19         for(j=0;j<p;j++){
20             scanf("%ld", &b[i][j]);
21         }
22     }
23     //A*B=C
24     for(i=0;i<m;i++){
25         for(j=0;j<p;j++){
26             for(k=0;k<n;k++){
27                 c[i][j]=c[i][j]+a[i][k]*b[k][j];
28             }
29         }
30     }
31 
32     printf("A*B= \n");
33     for(i=0;i<m;i++){
34         for(j=0;j<p;j++){
35             printf("%3ld",c[i][j]);
