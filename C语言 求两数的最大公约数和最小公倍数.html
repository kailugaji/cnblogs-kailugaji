 1 #include<stdio.h>
 2 //最大公约数
 3 int gys(int x,int y){
 4     int r;
 5     while(y!=0){
 6         r=x%y;
 7         x=y;
 8         y=r;
 9     }
10     return x;
11 }
12 //最小公倍数
13 int gbs(int x,int y){
14     int z;
15     z=x*y/gys(x,y);
16     return z;
17 }
